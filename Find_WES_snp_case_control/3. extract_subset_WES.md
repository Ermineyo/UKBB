#### Description
In this step, we will extract WES plink file subset from UKB.

#### Script
** submit on local computer, and download the results from UKB RAP.

```
#!/bin/bash

path="/Bulk/Exome sequences/Population level exome OQFE variants, PLINK format - final release"


for chr in {1..22} X Y
do

run_plink_case_subset="plink --bfile ukb23158_c${chr}_b0_v1 --keep case_keep_samples.txt --make-bed --out case_chr${chr}_subset"

run_plink_control_subset="plink --bfile ukb23158_c${chr}_b0_v1 --keep control_keep_samples.txt --make-bed --out control_chr${chr}_subset"

dx run swiss-army-knife -iin="${path}/ukb23158_c${chr}_b0_v1.bed" \
    -iin="${path}/ukb23158_c${chr}_b0_v1.bim"\
    -iin="${path}/ukb23158_c${chr}_b0_v1.fam"\
    -iin="/zhanyezhang/case_keep_samples.txt"\
    -icmd="${run_plink_case_subset}" --tag="Plink Subset Case Chr${chr}" --instance-type "mem1_ssd1_v2_x36"\
    --destination="/zhanyezhang/subset/" --brief --yes

dx run swiss-army-knife -iin="${path}/ukb23158_c${chr}_b0_v1.bed" \
    -iin="${path}/ukb23158_c${chr}_b0_v1.bim"\
    -iin="${path}/ukb23158_c${chr}_b0_v1.fam"\
    -iin="/zhanyezhang/control_keep_samples.txt"\
    -icmd="${run_plink_control_subset}" --tag="Plink Subset Control Chr${chr}" --instance-type "mem1_ssd1_v2_x36"\
    --destination="/zhanyezhang/subset/" --brief --yes

done

```
